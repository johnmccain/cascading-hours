169 function cascading_hours_import_parse($contents)
170 {
171  //parse file contents from string to multidimensional array
172  $errors = [];
173  $contents = explode("\n", $contents);
174  $schedule = [];
175  $prev_day = false;
176  foreach($contents as $key => &$row) {
177  $row = explode(",", $row);
178  array_map('cascading_hours_remove_encapsulating_quotes', $row);
179  if(count($row) % 2 != 1) {
180  //there should be an odd number of columns for each row (date + pairs of start/end times)
181  $errors[] = t("On row $key: incorrect number of columns. Each row should begin with a date and continue with pairs of start/end times.");
182  } else {
183  if(count($contents) - 1 == $key && $row[0] == "") {
184  //end of file line, skip but no error
185  continue;
186  } else if(strtotime($row[0]) == false) {
187  $errors[] = t("On row $key: unable to parse date.");
188  continue;
189  } else if($prev_day && strtotime($row[0]) - $prev_day > 90000) {
190  //if more than 1 day has passed since last day then mark an error (90000s = 25hrs to account for days with daylight savings time)
191  $errors[] = t("On row $key: day skipped. All days in import list should be sequential.");
192  }
193  $prev_day = strtotime($row[0]);
194  $blocks = [];
195  $tmp = array_slice($row, 1);
196  for($i = 0; $i + 1 < count($tmp); $i += 2) {
197  if($tmp[$i] == "") {
198  continue;
199  }
200  $block = [];
201  if(strtotime($tmp[$i]) == false || strtotime($tmp[$i + 1] == false)) {
202  $errors[] = t("On row $key: unable to parse time(s).");
203  continue;
204  }
205  $block['start_time'] = date("h:i a", strtotime($tmp[$i]));
206  $block['end_time'] = date("h:i a", strtotime($tmp[$i + 1]));
207  $blocks[] = $block;
208  }
209  $schedule[date("Y-m-d", strtotime($row[0]))] = $blocks;
210  }
211  }
212 
213  if($errors != []) {
214  $schedule = array('errors' => $errors);
215  }
216  return $schedule;
217 }
