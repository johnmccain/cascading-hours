565 function cascading_hours_generate_schedule($schedules, $start_date)
566 {
567  $obj = []; //schedule object (ready for output to json)
568  foreach($schedules as $key => $schedule) {
569  $date = new DateTime();
570  $date->setTimestamp($start_date + ($key * 86400));
571  $date_str = $date->format('Y-m-d ');
572  $schedule_blocks = []; //array of schedule blocks starting in the same day
573  foreach($schedule as $value) {
574  $block = []; //schedule block object with start and end datetimes
575  $block['start'] = DateTime::createFromFormat('Y-m-d H:i:s', $date_str . $value['start_time'])
576  ->format(DATE_ATOM);
577  $block['end'] = DateTime::createFromFormat('Y-m-d H:i:s', $date_str . $value['end_time'])
578  ->format(DATE_ATOM);
579  $schedule_blocks[] = $block;
580  }
581 
582  // sort by block start time
583 
584  usort($schedule_blocks, "cascading_hours_block_cmp");
585  $obj[] = $schedule_blocks;
586  }
587 
588  return $obj;
589 }
