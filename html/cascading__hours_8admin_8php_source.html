469 function cascading_hours_admin_render_rule_view($rules, $start_date, $end_date) {
470  $page = '';
471  $date_iterator = new DateTime();
472  $end = new DateTime();
473  $end->setTimestamp($end_date);
474  $page .= '<table><tr><th>Weight</th>';
475  for($date_iterator->setTimestamp($start_date); $date_iterator < $end; $date_iterator->modify('+1 day')) {
476  $page .= '<th>' . $date_iterator->format('m/d') . '</th>';
477  }
478  $page .= '</tr>';
479 
480  //iterate through priorities
481  for($i = -5; $i <= 5; $i++) {
482  $end->setTimestamp($end_date);
483  $page .= '<tr><th>' . $i . '</th>';
484  //iterate through dates
485  for($date_iterator->setTimestamp($start_date); $date_iterator < $end; $date_iterator->modify('+1 day')) {
486  $tmp_rule = false;
487  //find a rule that applies (if it exists)
488  foreach($rules as $rule) {
489  if($rule['priority'] == $i &&
490  strtotime($rule['start_date']) <= $date_iterator->getTimestamp()
491  && strtotime($rule['end_date']) >= $date_iterator->getTimestamp()) {
492  $tmp_rule = $rule;
493  break;
494  }
495  }
496  if($tmp_rule) {
497  $page .= '<td class="cascading_hours_grid_view_on">' . l($tmp_rule['alias'] . ' ' . $tmp_rule['id'], "admin/structure/cascading_hours/rule/" . $tmp_rule['id']) . '</td>';
498  } else {
499  $page .= '<td>&nbsp;</td>';
500  }
501  }
502  $page .= '</tr>';
503  }
504  $page .= '</table>';
505  return $page;
506 }
